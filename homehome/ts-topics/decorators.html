Decorators in TypeScript

Decorators are a special kind of declaration that can be attached to a class declaration, method, accessor, property, or parameter. They use the form @expression, where expression must evaluate to a function that will be called at runtime with information about the decorated declaration.

Experimental Feature

Decorators are an experimental feature in TypeScript. To enable them, you need to enable the experimentalDecorators compiler option in your tsconfig.json:

{
  "compilerOptions": {
    "target": "ES5",
    "experimentalDecorators": true
  }
}

Class Decorators

Class decorators are applied to the constructor of the class and can be used to observe, modify, or replace a class definition:

// Class decorator function
function sealed(constructor: Function) {
    Object.seal(constructor);
    Object.seal(constructor.prototype);
}

// Using the decorator
@sealed
class BugReport {
    type = "report";
    title: string;
    
    constructor(title: string) {
        this.title = title;
    }
}

// Decorator factory (returns a decorator function)
function classDecorator(
    constructor: T
) {
    return class extends constructor {
        newProperty = "new property";
        hello = "override";
    };
}

@classDecorator
class Greeter {
    property = "property";
    hello: string;
    
    constructor(m: string) {
        this.hello = m;
    }
}

console.log(new Greeter("world"));
// Output: { property: 'property', hello: 'override', newProperty: 'new property' }

Method Decorators

Method decorators are applied to a method of a class and can be used to observe, modify, or replace a method definition:

// Method decorator function
function log(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    // Save the original method
    const originalMethod = descriptor.value;
    
    // Replace the method with a new one that logs before and after
    descriptor.value = function(...args: any[]) {
        console.log(`Calling ${propertyKey} with args: ${JSON.stringify(args)}`);
        const result = originalMethod.apply(this, args);
        console.log(`Method ${propertyKey} returned: ${JSON.stringify(result)}`);
        return result;
    };
    
    return descriptor;
}

class Calculator {
    @log
    add(a: number, b: number) {
        return a + b;
    }
}

const calc = new Calculator();
calc.add(1, 2);
// Output:
// Calling add with args: [1,2]
// Method add returned: 3

Property Decorators

Property decorators are applied to a property of a class and can be used to observe or modify a property definition:

// Property decorator function
function format(formatString: string) {
    return function(target: any, propertyKey: string) {
        // Property value storage
        let value: string;
        
        // Create a getter and setter for the property
        const getter = function() {
            return value;
        };
        
        const setter = function(newValue: string) {
            value = formatString.replace("%s", newValue);
        };
        
        // Replace the property with getter and setter
        Object.defineProperty(target, propertyKey, {
            get: getter,
            set: setter,
            enumerable: true,
            configurable: true
        });
    };
}

class Greeter {
    @format("Hello, %s!")
    greeting: string;
}

const greeter = new Greeter();
greeter.greeting = "World";
console.log(greeter.greeting); // Output: Hello, World!

Parameter Decorators

Parameter decorators are applied to a parameter of a method or constructor and can be used to observe a parameter declaration:

// Parameter decorator function
function required(target: Object, propertyKey: string | symbol, parameterIndex: number) {
    // Get the existing design type metadata or create a new array
    const existingRequiredParameters: number[] = Reflect.getOwnMetadata(
        "required", target, propertyKey) || [];
    
    // Add the current parameter index to the required parameters list
    existingRequiredParameters.push(parameterIndex);
    
    // Update the metadata
    Reflect.defineMetadata(
        "required", existingRequiredParameters, target, propertyKey
    );
}

// Method decorator that validates required parameters
function validate(target: any, propertyName: string, descriptor: PropertyDescriptor) {
    const method = descriptor.value;
    
    descriptor.value = function(...args: any[]) {
        // Get the list of required parameters
        const requiredParameters: number[] = Reflect.getOwnMetadata(
            "required", target, propertyName) || [];
        
        // Check if all required parameters are provided and not null/undefined
        for (const index of requiredParameters) {
            if (index >= args.length || args[index] === undefined || args[index] === null) {
                throw new Error(`Missing required parameter at index ${index}`);
            }
        }
        
        return method.apply(this, args);
    };
    
    return descriptor;
}

class UserService {
    @validate
    createUser(@required name: string, @required email: string, age?: number) {
        return { name, email, age };
    }
}

const userService = new UserService();
userService.createUser("John", "john@example.com");  // OK
// userService.createUser(null, "john@example.com");  // Error: Missing required parameter
// userService.createUser("John", null);              // Error: Missing required parameter

Reflect Metadata

The example above uses the reflect-metadata library, which needs to be installed and imported:

npm install reflect-metadata
// Add to your main file:
import "reflect-metadata";

You also need to enable emitDecoratorMetadata in your tsconfig.json:

{
  "compilerOptions": {
    "experimentalDecorators": true,
    "emitDecoratorMetadata": true
  }
}

Accessor Decorators

Accessor decorators are applied to the property descriptor for an accessor and can be used to observe, modify, or replace an accessor's definitions:

// Accessor decorator function
function configurable(value: boolean) {
    return function(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        descriptor.configurable = value;
        return descriptor;
    };
}

class Point {
    private _x: number;
    private _y: number;
    
    constructor(x: number, y: number) {
        this._x = x;
        this._y = y;
    }
    
    @configurable(false)
    get x() {
        return this._x;
    }
    
    @configurable(false)
    get y() {
        return this._y;
    }
}

Decorator Evaluation Order

The order in which decorators are evaluated is important to understand:

  1. Parameter Decorators, followed by Method, Accessor, or Property Decorators are applied for each instance member.
  2. Parameter Decorators, followed by Method, Accessor, or Property Decorators are applied for each static member.
  3. Parameter Decorators are applied for the constructor.
  4. Class Decorators are applied for the class.

Real-World Use Cases

1. Dependency Injection

// Simple dependency injection system
const Injectable = (): ClassDecorator => {
    return (target: any) => {
        // Register the class in the DI container
        DI.register(target);
    };
};

@Injectable()
class UserService {
    getUsers() {
        return ["User1", "User2"];
    }
}

@Injectable()
class UserController {
    constructor(private userService: UserService) {}
    
    listUsers() {
        return this.userService.getUsers();
    }
}

2. Route Decorators (like in NestJS)

// Route decorator factory
function Route(path: string): MethodDecorator {
    return (target, propertyKey, descriptor) => {
        // Register the route
        Router.register(path, target[propertyKey]);
        return descriptor;
    };
}

// HTTP method decorators
const Get = (path: string) => Route(`GET ${path}`);
const Post = (path: string) => Route(`POST ${path}`);

class UserController {
    @Get("/users")
    getUsers() {
        return ["User1", "User2"];
    }
    
    @Post("/users")
    createUser(@Body() userData: any) {
        // Create user logic
        return { id: 1, ...userData };
    }
}

3. Validation (like in class-validator)

// Validation decorators
function IsString(target: any, propertyKey: string) {
    // Add validation metadata
    let validationRules = Reflect.getMetadata("validation", target) || {};
    validationRules[propertyKey] = validationRules[propertyKey] || [];
    validationRules[propertyKey].push({
        type: "string",
        validate: (value: any) => typeof value === "string"
    });
    Reflect.defineMetadata("validation", validationRules, target);
}

function IsEmail(target: any, propertyKey: string) {
    // Add validation metadata
    let validationRules = Reflect.getMetadata("validation", target) || {};
    validationRules[propertyKey] = validationRules[propertyKey] || [];
    validationRules[propertyKey].push({
        type: "email",
        validate: (value: any) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)
    });
    Reflect.defineMetadata("validation", validationRules, target);
}

// Usage
class CreateUserDto {
    @IsString
    name: string;
    
    @IsString
    @IsEmail
    email: string;
}

Decorator Metadata

Many of these examples rely on storing and retrieving metadata. In a real application, you would typically use the reflect-metadata library to handle this.

Decorators vs. Other Patterns

Decorators

  • Declarative syntax with @decorator
  • Applied at class definition time
  • Can modify class behavior
  • Metadata is stored with the class
  • Commonly used in frameworks

Higher-Order Functions

  • Functional approach
  • Applied at runtime
  • Wrap existing functionality
  • No built-in metadata storage
  • More flexible but less declarative
