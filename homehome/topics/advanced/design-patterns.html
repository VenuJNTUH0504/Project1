Design Patterns in C++

Understanding and implementing common design patterns in C++

Singleton Pattern

The Singleton pattern ensures a class has only one instance and provides a global point of access to it.

class Singleton {
private:
    static Singleton* instance;
    Singleton() {}  // Private constructor
    
public:
    static Singleton* getInstance() {
        if (!instance) {
            instance = new Singleton();
        }
        return instance;
    }
    
    // Delete copy constructor and assignment operator
    Singleton(const Singleton&) = delete;
    Singleton& operator=(const Singleton&) = delete;
    
    void doSomething() {
        // Implementation
    }
};

// Initialize static member
Singleton* Singleton::instance = nullptr;

// Usage
Singleton::getInstance()->doSomething();

Note: Modern C++ implementations often use local static variables for thread-safe initialization.

Factory Pattern

The Factory pattern provides an interface for creating objects in a superclass, but allows subclasses to alter the type of objects that will be created.

class Product {
public:
    virtual ~Product() = default;
    virtual void operation() = 0;
};

class ConcreteProductA : public Product {
public:
    void operation() override {
        // Implementation for Product A
    }
};

class ConcreteProductB : public Product {
public:
    void operation() override {
        // Implementation for Product B
    }
};

class Factory {
public:
    virtual std::unique_ptr createProduct() = 0;
    virtual ~Factory() = default;
};

class ConcreteFactoryA : public Factory {
public:
    std::unique_ptr createProduct() override {
        return std::make_unique();
    }
};

class ConcreteFactoryB : public Factory {
public:
    std::unique_ptr createProduct() override {
        return std::make_unique();
    }
};

Observer Pattern

The Observer pattern defines a one-to-many dependency between objects so that when one object changes state, all its dependents are notified and updated automatically.

#include 
#include 

class Subject {
private:
    std::vector> observers;
    
public:
    void attach(std::function observer) {
        observers.push_back(observer);
    }
    
    void notify() {
        for (const auto& observer : observers) {
            observer();
        }
    }
};

class ConcreteSubject : public Subject {
private:
    int state;
    
public:
    void setState(int newState) {
        state = newState;
        notify();
    }
    
    int getState() const {
        return state;
    }
};

class Observer {
public:
    virtual void update() = 0;
    virtual ~Observer() = default;
};

class ConcreteObserver : public Observer {
private:
    ConcreteSubject& subject;
    
public:
    ConcreteObserver(ConcreteSubject& s) : subject(s) {}
    
    void update() override {
        std::cout << "State changed to: " << subject.getState() << std::endl;
    }
};

Strategy Pattern

The Strategy pattern defines a family of algorithms, encapsulates each one, and makes them interchangeable.

class Strategy {
public:
    virtual void execute() = 0;
    virtual ~Strategy() = default;
};

class ConcreteStrategyA : public Strategy {
public:
    void execute() override {
        // Implementation of strategy A
    }
};

class ConcreteStrategyB : public Strategy {
public:
    void execute() override {
        // Implementation of strategy B
    }
};

class Context {
private:
    std::unique_ptr strategy;
    
public:
    void setStrategy(std::unique_ptr s) {
        strategy = std::move(s);
    }
    
    void executeStrategy() {
        if (strategy) {
            strategy->execute();
        }
    }
};

Best Practice: The Strategy pattern is often implemented using function objects or lambdas in modern C++.

Decorator Pattern

The Decorator pattern attaches additional responsibilities to an object dynamically.

class Component {
public:
    virtual void operation() = 0;
    virtual ~Component() = default;
};

class ConcreteComponent : public Component {
public:
    void operation() override {
        // Basic implementation
    }
};

class Decorator : public Component {
protected:
    std::unique_ptr component;
    
public:
    Decorator(std::unique_ptr c) : component(std::move(c)) {}
    
    void operation() override {
        if (component) {
            component->operation();
        }
    }
};

class ConcreteDecoratorA : public Decorator {
public:
    ConcreteDecoratorA(std::unique_ptr c) : Decorator(std::move(c)) {}
    
    void operation() override {
        Decorator::operation();
        // Additional behavior
    }
};

class ConcreteDecoratorB : public Decorator {
public:
    ConcreteDecoratorB(std::unique_ptr c) : Decorator(std::move(c)) {}
    
    void operation() override {
        Decorator::operation();
        // Additional behavior
    }
};

Command Pattern

The Command pattern encapsulates a request as an object, thereby letting you parameterize clients with different requests, queue or log requests, and support undoable operations.

class Command {
public:
    virtual void execute() = 0;
    virtual void undo() = 0;
    virtual ~Command() = default;
};

class Receiver {
public:
    void action() {
        // Perform the action
    }
    
    void undoAction() {
        // Undo the action
    }
};

class ConcreteCommand : public Command {
private:
    Receiver& receiver;
    
public:
    ConcreteCommand(Receiver& r) : receiver(r) {}
    
    void execute() override {
        receiver.action();
    }
    
    void undo() override {
        receiver.undoAction();
    }
};

class Invoker {
private:
    std::vector> commands;
    
public:
    void addCommand(std::unique_ptr cmd) {
        commands.push_back(std::move(cmd));
    }
    
    void executeCommands() {
        for (const auto& cmd : commands) {
            cmd->execute();
        }
    }
    
    void undoCommands() {
        for (auto it = commands.rbegin(); it != commands.rend(); ++it) {
            (*it)->undo();
        }
    }
};

Note: The Command pattern is particularly useful for implementing undo/redo functionality and macro recording.
