Introduction to Data Science

NumPy

Numerical computing with NumPy:

import numpy as np

# Create arrays
arr1 = np.array([1, 2, 3, 4, 5])
arr2 = np.array([[1, 2, 3], [4, 5, 6]])

# Array operations
print(arr1 + 2)  # [3, 4, 5, 6, 7]
print(arr1 * 2)  # [2, 4, 6, 8, 10]

# Statistical functions
print(np.mean(arr1))  # 3.0
print(np.std(arr1))   # 1.4142135623730951

# Linear algebra
matrix = np.array([[1, 2], [3, 4]])
print(np.linalg.det(matrix))  # -2.0

Pandas

Data manipulation with Pandas:

import pandas as pd

# Create DataFrame
data = {
    'Name': ['Alice', 'Bob', 'Charlie'],
    'Age': [25, 30, 35],
    'City': ['New York', 'London', 'Paris']
}
df = pd.DataFrame(data)

# Basic operations
print(df.head())
print(df.describe())

# Filtering
print(df[df['Age'] > 30])

# Grouping
print(df.groupby('City')['Age'].mean())

# Reading and writing data
df.to_csv('data.csv', index=False)
df = pd.read_csv('data.csv')

Matplotlib

Data visualization with Matplotlib:

import matplotlib.pyplot as plt
import numpy as np

# Create data
x = np.linspace(0, 10, 100)
y = np.sin(x)

# Create figure and axis
fig, ax = plt.subplots()

# Plot data
ax.plot(x, y, label='sin(x)')
ax.set_xlabel('x')
ax.set_ylabel('y')
ax.set_title('Sine Wave')
ax.legend()

# Save plot
plt.savefig('sine_wave.png')
plt.close()

Scikit-learn

Machine learning with Scikit-learn:

from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score

# Load data
iris = load_iris()
X, y = iris.data, iris.target

# Split data
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

# Train model
model = RandomForestClassifier()
model.fit(X_train, y_train)

# Make predictions
y_pred = model.predict(X_test)

# Evaluate model
accuracy = accuracy_score(y_test, y_pred)
print(f'Accuracy: {accuracy:.2f}')

TensorFlow

Deep learning with TensorFlow:

import tensorflow as tf
from tensorflow.keras import layers, models

# Create model
model = models.Sequential([
    layers.Dense(64, activation='relu', input_shape=(784,)),
    layers.Dense(64, activation='relu'),
    layers.Dense(10, activation='softmax')
])

# Compile model
model.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])

# Train model
model.fit(X_train, y_train, epochs=5, batch_size=32)

# Evaluate model
test_loss, test_acc = model.evaluate(X_test, y_test)
print(f'Test accuracy: {test_acc:.2f}')

Best Practices

  • Use virtual environments for project isolation
  • Document data preprocessing steps
  • Validate data quality and handle missing values
  • Use appropriate visualization techniques
  • Split data into training, validation, and test sets
  • Regularize models to prevent overfitting
  • Use cross-validation for model evaluation
  • Document model performance metrics
