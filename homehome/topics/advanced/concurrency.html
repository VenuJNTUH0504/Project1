Concurrency & Multithreading in C++

Mastering parallel programming and thread management in C++

Thread Management

Basic thread creation and management using the C++ standard library.

#include 
#include 

void threadFunction(int id) {
    std::cout << "Thread " << id << " is running\n";
}

int main() {
    std::thread t1(threadFunction, 1);
    std::thread t2(threadFunction, 2);
    
    t1.join();  // Wait for t1 to finish
    t2.join();  // Wait for t2 to finish
    
    return 0;
}

Warning: Always join or detach threads before they go out of scope to avoid program termination.

Mutex and Lock Guards

Protecting shared resources from concurrent access using mutexes.

#include 
#include 
#include 

std::mutex mtx;
int sharedData = 0;

void increment() {
    std::lock_guard lock(mtx);
    ++sharedData;
}

int main() {
    std::vector threads;
    for (int i = 0; i < 10; ++i) {
        threads.emplace_back(increment);
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    std::cout << "Final value: " << sharedData << std::endl;
    return 0;
}

Best Practice: Use RAII-style lock guards to ensure mutexes are always properly released.

Condition Variables

Synchronizing threads using condition variables for efficient waiting.

#include 
#include 
#include 
#include 

std::mutex mtx;
std::condition_variable cv;
std::queue dataQueue;
bool finished = false;

void producer() {
    for (int i = 0; i < 5; ++i) {
        std::lock_guard lock(mtx);
        dataQueue.push(i);
        cv.notify_one();
    }
    finished = true;
    cv.notify_all();
}

void consumer() {
    while (true) {
        std::unique_lock lock(mtx);
        cv.wait(lock, []{ return !dataQueue.empty() || finished; });
        
        if (finished && dataQueue.empty()) break;
        
        int value = dataQueue.front();
        dataQueue.pop();
        lock.unlock();
        
        std::cout << "Consumed: " << value << std::endl;
    }
}

Atomic Operations

Using atomic types for thread-safe operations without mutexes.

#include 
#include 
#include 

std::atomic counter(0);

void increment() {
    for (int i = 0; i < 1000; ++i) {
        counter.fetch_add(1, std::memory_order_relaxed);
    }
}

int main() {
    std::vector threads;
    for (int i = 0; i < 10; ++i) {
        threads.emplace_back(increment);
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    std::cout << "Final counter value: " << counter << std::endl;
    return 0;
}

Note: Atomic operations are generally faster than mutex-based synchronization but have more limited use cases.

Async and Futures

Asynchronous programming using std::async and std::future.

#include 
#include 
#include 

int compute(int x) {
    // Simulate some work
    std::this_thread::sleep_for(std::chrono::seconds(1));
    return x * x;
}

int main() {
    std::vector> futures;
    
    // Launch multiple async tasks
    for (int i = 1; i <= 5; ++i) {
        futures.push_back(std::async(std::launch::async, compute, i));
    }
    
    // Get results
    for (auto& f : futures) {
        std::cout << "Result: " << f.get() << std::endl;
    }
    
    return 0;
}

Thread Pools

Implementing a simple thread pool for efficient task execution.

#include 
#include 
#include 
#include 
#include 
#include 

class ThreadPool {
private:
    std::vector workers;
    std::queue> tasks;
    std::mutex queueMutex;
    std::condition_variable condition;
    bool stop = false;

public:
    ThreadPool(size_t threads) {
        for (size_t i = 0; i < threads; ++i) {
            workers.emplace_back([this] {
                while (true) {
                    std::function task;
                    {
                        std::unique_lock lock(queueMutex);
                        condition.wait(lock, [this] {
                            return stop || !tasks.empty();
                        });
                        if (stop && tasks.empty()) return;
                        task = std::move(tasks.front());
                        tasks.pop();
                    }
                    task();
                }
            });
        }
    }

    template
    void enqueue(F&& f) {
        {
            std::unique_lock lock(queueMutex);
            tasks.emplace(std::forward(f));
        }
        condition.notify_one();
    }

    ~ThreadPool() {
        {
            std::unique_lock lock(queueMutex);
            stop = true;
        }
        condition.notify_all();
        for (std::thread& worker : workers) {
            worker.join();
        }
    }
};

Warning: Thread pools require careful management of resources and task lifetimes.
