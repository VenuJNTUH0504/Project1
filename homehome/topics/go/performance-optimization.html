    // Pre-allocate result slice

CPU Optimization

Optimizing CPU usage:

// Using worker pools
func processWorkers(items []Item) []Result {
    numWorkers := runtime.NumCPU()
    results := make([]Result, len(items))
    
    var wg sync.WaitGroup
    wg.Add(numWorkers)
    
    workChan := make(chan int, len(items))
    for i := range items {
        workChan <- i
    }
    close(workChan)
    
    for i := 0; i < numWorkers; i++ {
        go func() {
            defer wg.Done()
            for idx := range workChan {
                results[idx] = processItem(items[idx])
            }
        }()
    }
    
    wg.Wait()
    return results
}

Concurrency Optimization

Optimizing concurrent operations:

// Using bounded concurrency
func processWithLimit(items []Item) []Result {
    results := make([]Result, len(items))
    sem := make(chan struct{}, 10) // Limit to 10 concurrent operations
    
    var wg sync.WaitGroup
    for i, item := range items {
        wg.Add(1)
        go func(idx int, it Item) {
            defer wg.Done()
            sem <- struct{}{}        // Acquire semaphore
            defer func() { <-sem }() // Release semaphore
            
            results[idx] = processItem(it)
        }(i, item)
    }
    
    wg.Wait()
    return results
}

// Using context for cancellation
func processWithTimeout(ctx context.Context, items []Item) ([]Result, error) {
    ctx, cancel := context.WithTimeout(ctx, 5*time.Second)
    defer cancel()
    
    results := make([]Result, len(items))
    errChan := make(chan error, 1)
    
    go func() {
        for i, item := range items {
            select {
            case <-ctx.Done():
                return
            default:
                results[i] = processItem(item)
            }
        }
        errChan <- nil
    }()
    
    select {
    case err := <-errChan:
        return results, err
    case <-ctx.Done():
        return nil, ctx.Err()
    }
}

I/O Optimization

Optimizing I/O operations:

// Using buffered I/O
func processFile(filename string) error {
    file, err := os.Open(filename)
    if err != nil {
        return err
    }
    defer file.Close()
    
    // Use buffered reader
    reader := bufio.NewReader(file)
    
    for {
        line, err := reader.ReadString('\n')
        if err != nil {
            if err == io.EOF {
                break
            }
            return err
        }
        // Process line
    }
    return nil
}

// Using connection pooling
var db *sql.DB

func init() {
    var err error
    db, err = sql.Open("postgres", "user=postgres dbname=mydb sslmode=disable")
    if err != nil {
        log.Fatal(err)
    }
    
    // Set connection pool parameters
    db.SetMaxOpenConns(25)
    db.SetMaxIdleConns(25)
    db.SetConnMaxLifetime(5 * time.Minute)
}
