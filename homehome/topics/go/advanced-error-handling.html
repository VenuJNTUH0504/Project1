Go Advanced Error Handling

Advanced patterns and techniques for error handling in Go

Error Wrapping

Using error wrapping for context:

// Basic error wrapping
func processFile(filename string) error {
    data, err := ioutil.ReadFile(filename)
    if err != nil {
        return fmt.Errorf("failed to read file %s: %w", filename, err)
    }
    
    // Process data
    return nil
}

// Using custom error types
type FileError struct {
    Path string
    Err  error
}

func (e *FileError) Error() string {
    return fmt.Sprintf("file error: %s: %v", e.Path, e.Err)
}

func (e *FileError) Unwrap() error {
    return e.Err
}

func processFile(filename string) error {
    data, err := ioutil.ReadFile(filename)
    if err != nil {
        return &FileError{
            Path: filename,
            Err:  err,
        }
    }
    
    // Process data
    return nil
}

Error Inspection

Inspecting and handling wrapped errors:

// Using errors.Is
if errors.Is(err, os.ErrNotExist) {
    // Handle file not found
}

// Using errors.As
var fileErr *FileError
if errors.As(err, &fileErr) {
    // Handle FileError
    fmt.Printf("Error with file %s: %v\n", fileErr.Path, fileErr.Err)
}

// Using error chains
func processData(data []byte) error {
    if err := validateData(data); err != nil {
        return fmt.Errorf("validation failed: %w", err)
    }
    
    if err := transformData(data); err != nil {
        return fmt.Errorf("transformation failed: %w", err)
    }
    
    return nil
}

Error Handling Patterns

Common error handling patterns:

// Error handling with context
func processRequest(r *http.Request) error {
    ctx := r.Context()
    
    // Add timeout
    ctx, cancel := context.WithTimeout(ctx, 5*time.Second)
    defer cancel()
    
    // Process with context
    if err := processWithContext(ctx); err != nil {
        return fmt.Errorf("request processing failed: %w", err)
    }
    
    return nil
}

// Error handling with recovery
func safeOperation() (err error) {
    defer func() {
        if r := recover(); r != nil {
            err = fmt.Errorf("panic recovered: %v", r)
        }
    }()
    
    // Operation that might panic
    return nil
}

Error Logging

Logging errors effectively:

// Structured error logging
type ErrorLog struct {
    Time    time.Time
    Level   string
    Message string
    Error   error
    Context map[string]interface{}
}

func logError(err error, context map[string]interface{}) {
    log := ErrorLog{
        Time:    time.Now(),
        Level:   "ERROR",
        Message: err.Error(),
        Error:   err,
        Context: context,
    }
    
    // Log to file or service
    logJSON, _ := json.Marshal(log)
    fmt.Println(string(logJSON))
}

// Using log levels
func handleError(err error) {
    if errors.Is(err, context.DeadlineExceeded) {
        log.Printf("WARNING: %v", err)
    } else {
        log.Printf("ERROR: %v", err)
    }
}

Error Metrics

Tracking and monitoring errors:

// Error metrics
type ErrorMetrics struct {
    Count     int
    LastError error
    Timestamp time.Time
}

var errorMetrics = make(map[string]*ErrorMetrics)

func trackError(category string, err error) {
    metrics, ok := errorMetrics[category]
    if !ok {
        metrics = &ErrorMetrics{}
        errorMetrics[category] = metrics
    }
    
    metrics.Count++
    metrics.LastError = err
    metrics.Timestamp = time.Now()
}

// Using Prometheus metrics
var errorCounter = prometheus.NewCounterVec(
    prometheus.CounterOpts{
        Name: "errors_total",
        Help: "Total number of errors by category",
    },
    []string{"category"},
)

func init() {
    prometheus.MustRegister(errorCounter)
}

func handleError(category string, err error) {
    errorCounter.WithLabelValues(category).Inc()
    // Handle error
}

Best Practices

  • Always wrap errors with context
  • Use custom error types for specific cases
  • Implement proper error unwrapping
  • Use errors.Is and errors.As for error inspection
  • Log errors with appropriate context
  • Track error metrics for monitoring
  • Handle panics gracefully
  • Use context for cancellation and timeouts
