Go Interfacing with C

Calling C code from Go and vice versa

CGo Basics

Basic CGo usage:

// Basic CGo example
package main

/*
#include 
#include 

void hello() {
    printf("Hello from C!\n");
}
*/
import "C"

func main() {
    C.hello()
}

// Using C types
/*
#include 

int32_t add(int32_t a, int32_t b) {
    return a + b;
}
*/
import "C"

func add(a, b int32) int32 {
    return int32(C.add(C.int32_t(a), C.int32_t(b)))
}

Memory Management

Handling memory between Go and C:

// Allocating C memory
/*
#include 
*/
import "C"
import "unsafe"

func allocateMemory(size int) unsafe.Pointer {
    return C.malloc(C.size_t(size))
}

func freeMemory(ptr unsafe.Pointer) {
    C.free(ptr)
}

// Using Go memory in C
/*
#include 

void processString(char* str) {
    strcat(str, " from C");
}
*/
import "C"
import "unsafe"

func processString(s string) string {
    cstr := C.CString(s)
    defer C.free(unsafe.Pointer(cstr))
    
    C.processString(cstr)
    return C.GoString(cstr)
}

C Structs

Working with C structs:

// Using C structs
/*
typedef struct {
    int x;
    int y;
} Point;

Point createPoint(int x, int y) {
    Point p = {x, y};
    return p;
}
*/
import "C"

type Point struct {
    X int32
    Y int32
}

func createPoint(x, y int32) Point {
    cPoint := C.createPoint(C.int(x), C.int(y))
    return Point{
        X: int32(cPoint.x),
        Y: int32(cPoint.y),
    }
}

// Passing Go structs to C
/*
typedef struct {
    int x;
    int y;
} Point;

void processPoint(Point* p) {
    p->x *= 2;
    p->y *= 2;
}
*/
import "C"
import "unsafe"

func processPoint(p *Point) {
    cPoint := (*C.Point)(unsafe.Pointer(p))
    C.processPoint(cPoint)
}

Callbacks

Using callbacks between Go and C:

// C calling Go functions
/*
typedef void (*Callback)(int);

void processWithCallback(int value, Callback callback) {
    callback(value * 2);
}
*/
import "C"

//export goCallback
func goCallback(value C.int) {
    fmt.Printf("Callback received: %d\n", value)
}

func main() {
    C.processWithCallback(42, C.Callback(C.goCallback))
}

// Go calling C callbacks
/*
typedef int (*Operation)(int, int);

int processOperation(int a, int b, Operation op) {
    return op(a, b);
}
*/
import "C"

func processOperation(a, b int, op func(int, int) int) int {
    // Convert Go function to C callback
    callback := func(a, b C.int) C.int {
        return C.int(op(int(a), int(b)))
    }
    
    return int(C.processOperation(C.int(a), C.int(b), C.Operation(callback)))
}

Best Practices

  • Minimize CGo calls for performance
  • Handle memory management carefully
  • Use appropriate types for C compatibility
  • Implement proper error handling
  • Use build tags for platform-specific code
  • Document CGo dependencies
  • Test CGo code thoroughly
  • Consider alternatives to CGo when possible
