JavaScript Objects

What are Objects?

Objects in JavaScript are collections of key-value pairs. They are used to store related data and functionality. Unlike arrays, which use numbered indices, objects use named properties to access their members.

Note: Almost everything in JavaScript is an object. Strings, numbers, and booleans can be objects (if defined with the new keyword). Arrays, functions, and dates are always objects.

Creating Objects

There are several ways to create objects in JavaScript:

1. Object Literal Notation

// Empty object
const emptyObject = {};

// Object with properties
const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30,
    email: "john.doe@example.com"
};

2. Object Constructor

// Empty object
const emptyObject = new Object();

// Adding properties
const person = new Object();
person.firstName = "John";
person.lastName = "Doe";
person.age = 30;
person.email = "john.doe@example.com";

3. Constructor Functions

// Constructor function
function Person(firstName, lastName, age, email) {
    this.firstName = firstName;
    this.lastName = lastName;
    this.age = age;
    this.email = email;
}

// Create new instances
const person1 = new Person("John", "Doe", 30, "john.doe@example.com");
const person2 = new Person("Jane", "Smith", 25, "jane.smith@example.com");

4. Object.create() Method

// Create an object with a specific prototype
const personProto = {
    getFullName: function() {
        return this.firstName + " " + this.lastName;
    }
};

const person = Object.create(personProto);
person.firstName = "John";
person.lastName = "Doe";
person.age = 30;

console.log(person.getFullName()); // "John Doe"

5. ES6 Classes

// ES6 Class
class Person {
    constructor(firstName, lastName, age, email) {
        this.firstName = firstName;
        this.lastName = lastName;
        this.age = age;
        this.email = email;
    }
    
    getFullName() {
        return this.firstName + " " + this.lastName;
    }
}

const person = new Person("John", "Doe", 30, "john.doe@example.com");
console.log(person.getFullName()); // "John Doe"

Accessing Object Properties

There are two main ways to access object properties:

1. Dot Notation

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

console.log(person.firstName); // "John"
console.log(person.age); // 30

2. Bracket Notation

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

console.log(person["firstName"]); // "John"
console.log(person["age"]); // 30

// Bracket notation is useful when property names are dynamic
const propertyName = "lastName";
console.log(person[propertyName]); // "Doe"
Best Practice: Use dot notation when you know the property name in advance. Use bracket notation when property names are dynamic or contain special characters.

Modifying Objects

Objects in JavaScript are mutable, which means you can change their properties after creation.

Adding Properties

const person = {
    firstName: "John",
    lastName: "Doe"
};

// Add new properties
person.age = 30;
person["email"] = "john.doe@example.com";

console.log(person);
// { firstName: "John", lastName: "Doe", age: 30, email: "john.doe@example.com" }

Modifying Properties

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

// Modify existing properties
person.age = 31;
person["lastName"] = "Smith";

console.log(person);
// { firstName: "John", lastName: "Smith", age: 31 }

Deleting Properties

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30,
    email: "john.doe@example.com"
};

// Delete a property
delete person.email;

console.log(person);
// { firstName: "John", lastName: "Doe", age: 30 }

Object Methods

Objects can contain functions as properties, which are called methods.

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30,
    
    // Method
    getFullName: function() {
        return this.firstName + " " + this.lastName;
    },
    
    // ES6 shorthand method syntax
    greet() {
        return `Hello, my name is ${this.firstName} ${this.lastName}`;
    }
};

console.log(person.getFullName()); // "John Doe"
console.log(person.greet()); // "Hello, my name is John Doe"
Note: Inside an object method, this refers to the object the method belongs to.

Object Properties and Methods

Checking if a Property Exists

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

// Using the in operator
console.log("firstName" in person); // true
console.log("email" in person); // false

// Using hasOwnProperty method
console.log(person.hasOwnProperty("firstName")); // true
console.log(person.hasOwnProperty("email")); // false

Getting Object Keys, Values, and Entries

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

// Get all keys
const keys = Object.keys(person);
console.log(keys); // ["firstName", "lastName", "age"]

// Get all values
const values = Object.values(person);
console.log(values); // ["John", "Doe", 30]

// Get all entries (key-value pairs)
const entries = Object.entries(person);
console.log(entries);
// [["firstName", "John"], ["lastName", "Doe"], ["age", 30]]

Object Destructuring

Object destructuring allows you to extract properties from objects and assign them to variables.

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30,
    address: {
        street: "123 Main St",
        city: "New York",
        country: "USA"
    }
};

// Basic destructuring
const { firstName, lastName, age } = person;
console.log(firstName); // "John"
console.log(lastName); // "Doe"
console.log(age); // 30

// Assigning to different variable names
const { firstName: fName, lastName: lName } = person;
console.log(fName); // "John"
console.log(lName); // "Doe"

// Default values
const { email = "default@example.com" } = person;
console.log(email); // "default@example.com"

// Nested destructuring
const { address: { city, country } } = person;
console.log(city); // "New York"
console.log(country); // "USA"

// Rest pattern
const { firstName: name, ...rest } = person;
console.log(name); // "John"
console.log(rest); // { lastName: "Doe", age: 30, address: {...} }

Spread Operator with Objects

The spread operator (...) allows you to create a shallow copy of an object or merge objects.

const person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

// Copy an object
const personCopy = { ...person };
console.log(personCopy); // { firstName: "John", lastName: "Doe", age: 30 }

// Merge objects
const details = {
    job: "Developer",
    company: "Tech Corp"
};

const personWithDetails = { ...person, ...details };
console.log(personWithDetails);
// { firstName: "John", lastName: "Doe", age: 30, job: "Developer", company: "Tech Corp" }

// Override properties
const updatedPerson = { ...person, age: 31, lastName: "Smith" };
console.log(updatedPerson);
// { firstName: "John", lastName: "Smith", age: 31 }

Object.assign()

The Object.assign() method copies all enumerable own properties from one or more source objects to a target object.

const person = {
    firstName: "John",
    lastName: "Doe"
};

// Copy an object
const personCopy = Object.assign({}, person);
console.log(personCopy); // { firstName: "John", lastName: "Doe" }

// Merge objects
const details = {
    age: 30,
    job: "Developer"
};

const personWithDetails = Object.assign({}, person, details);
console.log(personWithDetails);
// { firstName: "John", lastName: "Doe", age: 30, job: "Developer" }

// Modify an existing object
Object.assign(person, { age: 30, email: "john.doe@example.com" });
console.log(person);
// { firstName: "John", lastName: "Doe", age: 30, email: "john.doe@example.com" }

Object Prototypes and Inheritance

JavaScript uses prototypal inheritance, where objects can inherit properties and methods from other objects.

Prototype Chain

// Constructor function
function Person(firstName, lastName) {
    this.firstName = firstName;
    this.lastName = lastName;
}

// Add method to the prototype
Person.prototype.getFullName = function() {
    return this.firstName + " " + this.lastName;
};

const person = new Person("John", "Doe");
console.log(person.getFullName()); // "John Doe"

// Inheritance
function Employee(firstName, lastName, jobTitle) {
    // Call the parent constructor
    Person.call(this, firstName, lastName);
    this.jobTitle = jobTitle;
}

// Inherit from Person.prototype
Employee.prototype = Object.create(Person.prototype);
Employee.prototype.constructor = Employee;

// Add method to Employee.prototype
Employee.prototype.getJobInfo = function() {
    return this.getFullName() + ", " + this.jobTitle;
};

const employee = new Employee("Jane", "Smith", "Developer");
console.log(employee.getFullName()); // "Jane Smith"
console.log(employee.getJobInfo()); // "Jane Smith, Developer"

ES6 Class Inheritance

// Parent class
class Person {
    constructor(firstName, lastName) {
        this.firstName = firstName;
        this.lastName = lastName;
    }
    
    getFullName() {
        return this.firstName + " " + this.lastName;
    }
}

// Child class
class Employee extends Person {
    constructor(firstName, lastName, jobTitle) {
        super(firstName, lastName);
        this.jobTitle = jobTitle;
    }
    
    getJobInfo() {
        return this.getFullName() + ", " + this.jobTitle;
    }
}

const employee = new Employee("Jane", "Smith", "Developer");
console.log(employee.getFullName()); // "Jane Smith"
console.log(employee.getJobInfo()); // "Jane Smith, Developer"
