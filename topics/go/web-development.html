HTTP Server

Creating a basic HTTP server:

// Basic HTTP server
package main

import (
    "fmt"
    "net/http"
)

func handler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hello, %s!", r.URL.Path[1:])
}

func main() {
    http.HandleFunc("/", handler)
    http.ListenAndServe(":8080", nil)
}

// Using middleware
func loggingMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        fmt.Printf("Request: %s %s\n", r.Method, r.URL.Path)
        next.ServeHTTP(w, r)
    })
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("/", handler)
    http.ListenAndServe(":8080", loggingMiddleware(mux))
}

Routing

Using popular routing packages:

// Using gorilla/mux
import "github.com/gorilla/mux"

func main() {
    r := mux.NewRouter()
    
    // Basic routes
    r.HandleFunc("/", HomeHandler)
    r.HandleFunc("/products", ProductsHandler)
    r.HandleFunc("/articles", ArticlesHandler)
    
    // Path parameters
    r.HandleFunc("/products/{id}", ProductHandler)
    
    // Query parameters
    r.HandleFunc("/search", SearchHandler).Queries("q", "{query}")
    
    // Subrouters
    s := r.PathPrefix("/api").Subrouter()
    s.HandleFunc("/users", UsersHandler)
    
    http.ListenAndServe(":8080", r)
}

// Using chi
import "github.com/go-chi/chi"

func main() {
    r := chi.NewRouter()
    
    // Middleware
    r.Use(middleware.Logger)
    r.Use(middleware.Recoverer)
    
    // Routes
    r.Get("/", HomeHandler)
    r.Route("/api", func(r chi.Router) {
        r.Get("/users", UsersHandler)
        r.Post("/users", CreateUserHandler)
    })
    
    http.ListenAndServe(":8080", r)
}

Database Integration

Working with databases:

// Using database/sql
import (
    "database/sql"
    _ "github.com/lib/pq"
)

func main() {
    db, err := sql.Open("postgres", "user=postgres dbname=mydb sslmode=disable")
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()
    
    // Query
    rows, err := db.Query("SELECT id, name FROM users")
    if err != nil {
        log.Fatal(err)
    }
    defer rows.Close()
    
    for rows.Next() {
        var id int
        var name string
        if err := rows.Scan(&id, &name); err != nil {
            log.Fatal(err)
        }
        fmt.Printf("%d: %s\n", id, name)
    }
}

// Using GORM
import "gorm.io/gorm"

type User struct {
    gorm.Model
    Name  string
    Email string
}

func main() {
    db, err := gorm.Open(postgres.Open("user=postgres dbname=mydb sslmode=disable"), &gorm.Config{})
    if err != nil {
        log.Fatal(err)
    }
    
    // Auto migrate
    db.AutoMigrate(&User{})
    
    // Create
    user := User{Name: "John", Email: "john@example.com"}
    db.Create(&user)
    
    // Query
    var users []User
    db.Find(&users)
}

Authentication

Implementing authentication:

// Using JWT
import "github.com/golang-jwt/jwt"

func generateToken(user User) (string, error) {
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, jwt.MapClaims{
        "user_id": user.ID,
        "exp":     time.Now().Add(time.Hour * 24).Unix(),
    })
    
    return token.SignedString([]byte("secret"))
}

// Using middleware
func authMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        tokenString := r.Header.Get("Authorization")
        if tokenString == "" {
            http.Error(w, "Unauthorized", http.StatusUnauthorized)
            return
        }
        
        token, err := jwt.Parse(tokenString, func(token *jwt.Token) (interface{}, error) {
            return []byte("secret"), nil
        })
        
        if err != nil || !token.Valid {
            http.Error(w, "Unauthorized", http.StatusUnauthorized)
            return
        }
        
        next.ServeHTTP(w, r)
    })
}
