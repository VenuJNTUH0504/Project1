Context

Using context for cancellation and deadlines:

// Creating context with timeout
ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
defer cancel()

// Using context in goroutines
go func(ctx context.Context) {
    select {
    case <-ctx.Done():
        fmt.Println("Operation cancelled")
        return
    case result := <-operation():
        fmt.Println("Operation completed:", result)
    }
}(ctx)

// Context with values
ctx := context.WithValue(context.Background(), "key", "value")
value := ctx.Value("key").(string)

Worker Pools

Implementing worker pools for controlled concurrency:

// Worker pool implementation
type WorkerPool struct {
    tasks chan Task
    wg    sync.WaitGroup
}

func NewWorkerPool(numWorkers int) *WorkerPool {
    pool := &WorkerPool{
        tasks: make(chan Task, 100),
    }
    pool.wg.Add(numWorkers)
    for i := 0; i < numWorkers; i++ {
        go pool.worker()
    }
    return pool
}

func (p *WorkerPool) worker() {
    defer p.wg.Done()
    for task := range p.tasks {
        task.Execute()
    }
}

func (p *WorkerPool) Submit(task Task) {
    p.tasks <- task
}

func (p *WorkerPool) Shutdown() {
    close(p.tasks)
    p.wg.Wait()}

Rate Limiting

Implementing rate limiting for controlled access:

// Token bucket rate limiter
type RateLimiter struct {
    tokens chan struct{}
    ticker *time.Ticker
}

func NewRateLimiter(rate int, interval time.Duration) *RateLimiter {
    rl := &RateLimiter{
        tokens: make(chan struct{}, rate),
        ticker: time.NewTicker(interval),
    }
    go rl.fill()
    return rl
}

func (rl *RateLimiter) fill() {
    for range rl.ticker.C {
        select {
        case rl.tokens <- struct{}{}:
        default:
        }
    }
}

func (rl *RateLimiter) Allow() bool {
    select {
    case <-rl.tokens:
        return true
    default:
        return false
    }
}

Circuit Breaker

Implementing circuit breaker pattern:

// Circuit breaker implementation
type CircuitBreaker struct {
    threshold    int
    failures     int
    resetTimeout time.Duration
    lastFailure  time.Time
    mu           sync.Mutex
}

func NewCircuitBreaker(threshold int, resetTimeout time.Duration) *CircuitBreaker {
    return &CircuitBreaker{
        threshold:    threshold,
        resetTimeout: resetTimeout,
    }
}

func (cb *CircuitBreaker) Execute(fn func() error) error {
    cb.mu.Lock()
    if cb.failures >= cb.threshold {
        if time.Since(cb.lastFailure) < cb.resetTimeout {
            cb.mu.Unlock()
            return fmt.Errorf("circuit breaker is open")
        }
        cb.failures = 0
    }
    cb.mu.Unlock()

    err := fn()
    if err != nil {
        cb.mu.Lock()
        cb.failures++
        cb.lastFailure = time.Now()
        cb.mu.Unlock()
    }
    return err
}

Concurrent Maps

Using sync.Map for concurrent map access:

// Using sync.Map
var m sync.Map

// Storing values
m.Store("key", "value")

// Loading values
if value, ok := m.Load("key"); ok {
    fmt.Println(value)
}

// Deleting values
m.Delete("key")

// Range over map
m.Range(func(key, value interface{}) bool {
    fmt.Println(key, value)
    return true
})

Best Practices

  • Use context for cancellation and timeouts
  • Implement proper error handling in goroutines
  • Use worker pools for controlled concurrency
  • Implement rate limiting for external services
  • Use circuit breakers for fault tolerance
  • Prefer sync.Map for concurrent map access
  • Use proper synchronization primitives
  • Monitor goroutine leaks
