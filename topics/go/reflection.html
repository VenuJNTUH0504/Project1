Go Reflection

Understanding and using reflection in Go

Reflection Basics

Basic reflection operations in Go:

// Getting type information
var x float64 = 3.4
t := reflect.TypeOf(x)
v := reflect.ValueOf(x)

// Type and value information
fmt.Println("Type:", t)
fmt.Println("Value:", v)
fmt.Println("Kind:", v.Kind())
fmt.Println("Float value:", v.Float())

// Setting values
var y float64 = 3.4
v = reflect.ValueOf(&y).Elem()
v.SetFloat(7.1)
fmt.Println("New value:", y)

Struct Reflection

Reflecting on struct types and values:

// Struct reflection
type Person struct {
    Name string
    Age  int
}

p := Person{"Alice", 25}
v := reflect.ValueOf(p)

// Getting field information
for i := 0; i < v.NumField(); i++ {
    field := v.Type().Field(i)
    value := v.Field(i)
    fmt.Printf("%s: %v\n", field.Name, value)
}

// Setting struct fields
v = reflect.ValueOf(&p).Elem()
v.FieldByName("Age").SetInt(30)
fmt.Println("Updated person:", p)

Function Reflection

Reflecting on functions:

// Function reflection
func Add(a, b int) int {
    return a + b
}

// Getting function information
f := reflect.ValueOf(Add)
fmt.Println("Function type:", f.Type())

// Calling functions through reflection
args := []reflect.Value{
    reflect.ValueOf(10),
    reflect.ValueOf(20),
}
result := f.Call(args)
fmt.Println("Result:", result[0].Int())

Interface Reflection

Reflecting on interfaces:

// Interface reflection
var i interface{} = "hello"
v := reflect.ValueOf(i)

// Checking interface type
if v.Kind() == reflect.String {
    fmt.Println("String value:", v.String())
}

// Type assertions with reflection
if v.Type().AssignableTo(reflect.TypeOf("")) {
    s := v.Interface().(string)
    fmt.Println("Converted string:", s)
}

Custom Tags

Using struct tags with reflection:

// Struct with tags
type User struct {
    Name string `json:"name" validate:"required"`
    Age  int    `json:"age" validate:"min=18"`
}

// Reading tags
t := reflect.TypeOf(User{})
field, _ := t.FieldByName("Name")
fmt.Println("JSON tag:", field.Tag.Get("json"))
fmt.Println("Validate tag:", field.Tag.Get("validate"))

// Processing tags
func ProcessTags(v interface{}) {
    t := reflect.TypeOf(v)
    for i := 0; i < t.NumField(); i++ {
        field := t.Field(i)
        if tag := field.Tag.Get("validate"); tag != "" {
            fmt.Printf("Field %s has validation: %s\n", field.Name, tag)
        }
    }
}

Best Practices

  • Use reflection sparingly and only when necessary
  • Prefer type assertions over reflection when possible
  • Handle panics from reflection operations
  • Cache reflection results for performance
  • Use struct tags for metadata
  • Document reflection usage clearly
  • Consider alternatives to reflection
  • Test reflection code thoroughly
