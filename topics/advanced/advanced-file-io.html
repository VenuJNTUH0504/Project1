Advanced File I/O in C++

Mastering file operations and stream handling in C++

File Streams and Modes

C++ provides different file stream classes and modes for various file operations.

#include 
#include 

int main() {
    // Open file for writing (truncate if exists)
    std::ofstream outFile("data.txt", std::ios::out | std::ios::trunc);
    
    // Open file for reading and writing (preserve content)
    std::fstream inOutFile("data.txt", 
        std::ios::in | std::ios::out | std::ios::app);
    
    // Open file for binary operations
    std::ofstream binaryFile("data.bin", 
        std::ios::binary | std::ios::out);
    
    // Check if file is open
    if (outFile.is_open()) {
        outFile << "Hello, World!" << std::endl;
        outFile.close();
    }
    
    return 0;
}

Note: Always check if the file is successfully opened before performing operations.

Binary File Operations

Binary file operations allow direct reading and writing of raw data.

struct Record {
    int id;
    char name[50];
    double value;
};

void writeBinary(const std::string& filename) {
    std::ofstream file(filename, std::ios::binary);
    if (file) {
        Record record = {1, "Test", 3.14};
        file.write(reinterpret_cast(&record), 
                  sizeof(Record));
    }
}

void readBinary(const std::string& filename) {
    std::ifstream file(filename, std::ios::binary);
    if (file) {
        Record record;
        file.read(reinterpret_cast(&record), 
                 sizeof(Record));
    }
}

Warning: Binary operations require careful handling of data alignment and endianness.

File Position and Seeking

Controlling file position and seeking to specific locations in a file.

void filePositioning() {
    std::fstream file("data.txt", 
        std::ios::in | std::ios::out);
    
    if (file) {
        // Get current position
        std::streampos pos = file.tellg();
        
        // Seek to beginning
        file.seekg(0, std::ios::beg);
        
        // Seek to end
        file.seekg(0, std::ios::end);
        
        // Get file size
        std::streampos size = file.tellg();
        
        // Seek to specific position
        file.seekg(10, std::ios::beg);
        
        // Relative seeking
        file.seekg(5, std::ios::cur);
    }
}

String Streams

String streams provide in-memory string manipulation with stream-like operations.

#include 
#include 

void stringStreamExample() {
    // Writing to string stream
    std::ostringstream oss;
    oss << "Value: " << 42 << " Pi: " << 3.14159;
    std::string result = oss.str();
    
    // Reading from string stream
    std::istringstream iss("100 3.14 Hello");
    int i;
    double d;
    std::string s;
    iss >> i >> d >> s;
    
    // String stream for formatting
    std::stringstream ss;
    ss << std::hex << 255;  // Converts to "ff"
    ss << std::setprecision(2) << 3.14159;  // "3.14"
}

Best Practice: String streams are useful for type conversion and string formatting.

File System Operations (C++17)

Modern file system operations using the filesystem library.

#include 
namespace fs = std::filesystem;

void fileSystemOperations() {
    // Create directory
    fs::create_directory("new_folder");
    
    // Check if path exists
    if (fs::exists("file.txt")) {
        // Get file size
        auto size = fs::file_size("file.txt");
        
        // Get last write time
        auto time = fs::last_write_time("file.txt");
        
        // Copy file
        fs::copy_file("source.txt", "destination.txt");

        fs::remove("file.txt");

        for (const auto& entry : fs::directory_iterator(".")) {
            std::cout << entry.path() << std::endl;
        }
    }
}

Memory-Mapped Files

Using memory-mapped files for efficient large file handling.

#include 
#include 
#include 

class MemoryMappedFile {
private:
    int fd;
    void* mapped;
    size_t size;
    
public:
    MemoryMappedFile(const std::string& filename) {
        fd = open(filename.c_str(), O_RDWR);
        if (fd == -1) return;
        
        size = lseek(fd, 0, SEEK_END);
        lseek(fd, 0, SEEK_SET);
        
        mapped = mmap(nullptr, size, 
            PROT_READ | PROT_WRITE, 
            MAP_SHARED, fd, 0);
    }
    
    ~MemoryMappedFile() {
        if (mapped) munmap(mapped, size);
        if (fd != -1) close(fd);
    }
    
    void* data() const { return mapped; }
    size_t length() const { return size; }
};

Warning: Memory-mapped files require careful handling of system resources and synchronization.

Asynchronous File I/O

Performing asynchronous file operations using futures and async.

#include 
#include 
#include 

std::future> readFileAsync(
    const std::string& filename) {
    return std::async(std::launch::async, 
        [filename]() {
            std::ifstream file(filename, 
                std::ios::binary | std::ios::ate);
            if (!file) return std::vector();
            
            auto size = file.tellg();
            file.seekg(0);
            
            std::vector buffer(size);
            file.read(buffer.data(), size);
            
            return buffer;
        });
}

void useAsyncIO() {
    auto future = readFileAsync("large_file.bin");
    // Do other work while file is being read
    
    // Get the result when needed
    auto data = future.get();
}

Best Practice: Asynchronous I/O is particularly useful for handling large files or network operations.
