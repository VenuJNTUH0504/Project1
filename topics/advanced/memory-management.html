Memory Management in C++

Understanding and implementing efficient memory management techniques

Smart Pointers

Smart pointers provide automatic memory management and prevent memory leaks.

#include 

// Unique pointer (exclusive ownership)
std::unique_ptr uptr = std::make_unique(42);

// Shared pointer (shared ownership)
std::shared_ptr sptr = std::make_shared(42);

// Weak pointer (non-owning reference)
std::weak_ptr wptr = sptr;

// Custom deleter
auto deleter = [](int* p) {
    std::cout << "Deleting " << *p << std::endl;
    delete p;
};
std::unique_ptr customPtr(new int(42), deleter);

Best Practice: Prefer smart pointers over raw pointers for automatic memory management.

Custom Memory Allocators

Custom memory allocators allow you to control how memory is allocated and deallocated.

class CustomAllocator {
private:
    static const size_t BLOCK_SIZE = 4096;
    struct Block {
        Block* next;
        char data[BLOCK_SIZE];
    };
    Block* currentBlock = nullptr;
    size_t currentOffset = 0;

public:
    void* allocate(size_t size) {
        if (currentBlock == nullptr || 
            currentOffset + size > BLOCK_SIZE) {
            Block* newBlock = new Block;
            newBlock->next = currentBlock;
            currentBlock = newBlock;
            currentOffset = 0;
        }
        void* ptr = ¤tBlock->data[currentOffset];
        currentOffset += size;
        return ptr;
    }

    void deallocate() {
        while (currentBlock) {
            Block* next = currentBlock->next;
            delete currentBlock;
            currentBlock = next;
        }
        currentOffset = 0;
    }
};

Warning: Custom allocators must be carefully implemented to avoid memory leaks and fragmentation.

Memory Pools

Memory pools improve performance by pre-allocating memory for objects of the same size.

template 
class MemoryPool {
private:
    struct Node {
        Node* next;
    };
    Node* freeList = nullptr;
    std::vector blocks;
    static const size_t BLOCK_SIZE = 1024;

public:
    T* allocate() {
        if (!freeList) {
            T* newBlock = new T[BLOCK_SIZE];
            blocks.push_back(newBlock);
            
            for (size_t i = 0; i < BLOCK_SIZE; ++i) {
                Node* node = reinterpret_cast(&newBlock[i]);
                node->next = freeList;
                freeList = node;
            }
        }
        
        Node* node = freeList;
        freeList = freeList->next;
        return reinterpret_cast(node);
    }

    void deallocate(T* ptr) {
        Node* node = reinterpret_cast(ptr);
        node->next = freeList;
        freeList = node;
    }

    ~MemoryPool() {
        for (T* block : blocks) {
            delete[] block;
        }
    }
};

Placement New

Placement new allows you to construct objects in pre-allocated memory.

class Object {
public:
    Object(int value) : data(value) {}
    int data;
};

// Pre-allocate memory
char buffer[sizeof(Object)];

// Construct object in pre-allocated memory
Object* obj = new (buffer) Object(42);

// Explicitly call destructor
obj->~Object();

Note: Placement new is useful for custom memory management and implementing containers.

Memory Alignment

Proper memory alignment is crucial for performance and hardware compatibility.

#include 

// Aligned structure
struct alignas(16) AlignedStruct {
    int32_t a;
    int64_t b;
    char c;
};

// Dynamic alignment
void* allocateAligned(size_t size, size_t alignment) {
    void* ptr = nullptr;
    if (posix_memalign(&ptr, alignment, size) != 0) {
        throw std::bad_alloc();
    }
    return ptr;
}

// Usage
void* alignedMemory = allocateAligned(1024, 64);

Warning: Misaligned memory access can cause performance penalties or hardware exceptions.

Memory Leak Detection

Techniques for detecting and preventing memory leaks.

#ifdef _DEBUG
class MemoryTracker {
private:
    static std::map allocations;
    static std::mutex mutex;

public:
    static void* allocate(size_t size) {
        void* ptr = malloc(size);
        std::lock_guard lock(mutex);
        allocations[ptr] = size;
        return ptr;
    }

    static void deallocate(void* ptr) {
        std::lock_guard lock(mutex);
        allocations.erase(ptr);
        free(ptr);
    }

    static void reportLeaks() {
        std::lock_guard lock(mutex);
        for (const auto& [ptr, size] : allocations) {
            std::cerr << "Memory leak: " << size 
                      << " bytes at " << ptr << std::endl;
        }
    }
};

// Override global new/delete
void* operator new(size_t size) {
    return MemoryTracker::allocate(size);
}

void operator delete(void* ptr) noexcept {
    MemoryTracker::deallocate(ptr);
}
#endif

Best Practice: Use memory leak detection tools during development and testing.
