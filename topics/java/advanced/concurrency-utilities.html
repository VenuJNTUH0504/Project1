Concurrency Utilities

Advanced concurrency features in Java

Introduction to Concurrency Utilities

Java provides a rich set of concurrency utilities in the java.util.concurrent package.

Key concurrency utilities:

  • Executor Framework
  • Atomic Variables
  • Locks and Conditions
  • Concurrent Collections
  • Synchronizers
  • Fork/Join Framework

Executor Framework

The Executor framework provides a higher-level replacement for working with threads directly.

Example: Executor Framework

import java.util.concurrent.*;

public class ExecutorExample {
    public static void main(String[] args) {
        // Create different types of executors
        ExecutorService fixedThreadPool = Executors.newFixedThreadPool(4);
        ExecutorService cachedThreadPool = Executors.newCachedThreadPool();
        ScheduledExecutorService scheduledExecutor = Executors.newScheduledThreadPool(2);
        
        // Submit tasks
        fixedThreadPool.submit(() -> {
            System.out.println("Task running in fixed thread pool");
        });
        
        // Schedule tasks
        scheduledExecutor.schedule(() -> {
            System.out.println("Delayed task");
        }, 1, TimeUnit.SECONDS);
        
        // Schedule periodic tasks
        scheduledExecutor.scheduleAtFixedRate(() -> {
            System.out.println("Periodic task");
        }, 0, 1, TimeUnit.SECONDS);
        
        // Shutdown executors
        fixedThreadPool.shutdown();
        cachedThreadPool.shutdown();
        scheduledExecutor.shutdown();
    }
}

Atomic Variables

Atomic variables provide thread-safe operations on single variables.

Example: Atomic Variables

import java.util.concurrent.atomic.*;

public class AtomicExample {
    public static void main(String[] args) {
        // AtomicInteger
        AtomicInteger counter = new AtomicInteger(0);
        counter.incrementAndGet();
        counter.addAndGet(5);
        
        // AtomicReference
        AtomicReference message = new AtomicReference<>("Hello");
        message.compareAndSet("Hello", "World");
        
        // AtomicLong
        AtomicLong timestamp = new AtomicLong(System.currentTimeMillis());
        timestamp.updateAndGet(x -> x + 1000);
        
        // AtomicBoolean
        AtomicBoolean flag = new AtomicBoolean(false);
        flag.compareAndSet(false, true);
        
        // AtomicArray
        AtomicIntegerArray array = new AtomicIntegerArray(10);
        array.incrementAndGet(0);
    }
}

Locks and Conditions

Locks provide more flexible synchronization than synchronized blocks.

Example: ReentrantLock and Condition

import java.util.concurrent.locks.*;

public class LockExample {
    private final Lock lock = new ReentrantLock();
    private final Condition notEmpty = lock.newCondition();
    private final Condition notFull = lock.newCondition();
    private final Queue queue = new LinkedList<>();
    private final int capacity = 10;

    public void produce(String item) throws InterruptedException {
        lock.lock();
        try {
            while (queue.size() == capacity) {
                notFull.await();
            }
            queue.add(item);
            notEmpty.signal();
        } finally {
            lock.unlock();
        }
    }

    public String consume() throws InterruptedException {
        lock.lock();
        try {
            while (queue.isEmpty()) {
                notEmpty.await();
            }
            String item = queue.remove();
            notFull.signal();
            return item;
        } finally {
            lock.unlock();
        }
    }
}

Concurrent Collections

Thread-safe collections for concurrent programming.

Example: Concurrent Collections

import java.util.concurrent.*;

public class ConcurrentCollectionsExample {
    public static void main(String[] args) {
        // ConcurrentHashMap
        ConcurrentHashMap map = new ConcurrentHashMap<>();
        map.put("one", 1);
        map.computeIfAbsent("two", k -> 2);
        
        // CopyOnWriteArrayList
        CopyOnWriteArrayList list = new CopyOnWriteArrayList<>();
        list.add("item1");
        list.addIfAbsent("item2");
        
        // BlockingQueue
        BlockingQueue queue = new LinkedBlockingQueue<>(10);
        try {
            queue.put("message1");
            String message = queue.take();
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        // ConcurrentSkipListSet
        ConcurrentSkipListSet set = new ConcurrentSkipListSet<>();
        set.add(1);
        set.add(2);
    }
}

Synchronizers

Synchronizers coordinate the control flow of threads.

Example: CountDownLatch and CyclicBarrier

import java.util.concurrent.*;

public class SynchronizersExample {
    public static void main(String[] args) throws InterruptedException {
        // CountDownLatch
        CountDownLatch latch = new CountDownLatch(3);
        
        for (int i = 0; i < 3; i++) {
            new Thread(() -> {
                System.out.println("Thread working");
                latch.countDown();
            }).start();
        }
        
        latch.await();
        System.out.println("All threads completed");
        
        // CyclicBarrier
        CyclicBarrier barrier = new CyclicBarrier(3, () -> 
            System.out.println("All threads reached barrier"));
        
        for (int i = 0; i < 3; i++) {
            new Thread(() -> {
                try {
                    System.out.println("Thread waiting at barrier");
                    barrier.await();
                } catch (Exception e) {
                    e.printStackTrace();
                }
            }).start();
        }
    }
}

Fork/Join Framework

The Fork/Join framework is designed for parallel processing of tasks.

Example: ForkJoinPool

import java.util.concurrent.*;

public class ForkJoinExample {
    static class SumTask extends RecursiveTask {
        private final long[] array;
        private final int start;
        private final int end;
        
        SumTask(long[] array, int start, int end) {
            this.array = array;
            this.start = start;
            this.end = end;
        }
        
        @Override
        protected Long compute() {
            if (end - start <= 1000) {
                long sum = 0;
                for (int i = start; i < end; i++) {
                    sum += array[i];
                }
                return sum;
            }
            
            int mid = (start + end) / 2;
            SumTask left = new SumTask(array, start, mid);
            SumTask right = new SumTask(array, mid, end);
            
            left.fork();
            long rightResult = right.compute();
            long leftResult = left.join();
            
            return leftResult + rightResult;
        }
    }
    
    public static void main(String[] args) {
        long[] array = new long[10000];
        for (int i = 0; i < array.length; i++) {
            array[i] = i;
        }
        
        ForkJoinPool pool = new ForkJoinPool();
        long sum = pool.invoke(new SumTask(array, 0, array.length));
        System.out.println("Sum: " + sum);
    }
}

Concurrency Best Practices

Best practices for concurrent programming:

  • Use higher-level concurrency utilities instead of low-level synchronization
  • Prefer immutable objects when possible
  • Use thread pools instead of creating new threads
  • Document thread-safety guarantees
  • Use concurrent collections instead of synchronized collections
  • Handle InterruptedException properly
  • Use appropriate synchronization granularity
  • Consider using CompletableFuture for asynchronous programming
