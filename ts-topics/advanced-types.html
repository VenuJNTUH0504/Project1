Advanced Types in TypeScript

TypeScript's type system goes beyond basic types like string, number, and boolean. Advanced types provide powerful tools for modeling complex data structures and relationships, making your code more expressive and type-safe.

Intersection Types

Intersection types combine multiple types into one, allowing an object to have all properties from each type:

interface BusinessPartner {
    name: string;
    credit: number;
}

interface Identity {
    id: number;
    email: string;
}

interface Contact {
    phone: string;
    address: string;
}

// Intersection type
type Employee = Identity & Contact;
type Customer = BusinessPartner & Contact;

// Usage
const employee: Employee = {
    id: 100,
    email: "john@example.com",
    phone: "555-1234",
    address: "123 Main St"
};

const customer: Customer = {
    name: "Acme Inc",
    credit: 10000,
    phone: "555-6789",
    address: "456 Business Ave"
};

Combining with Union Types

You can combine intersection and union types for complex type relationships:

type BusinessContact = BusinessPartner & Contact;
type PersonalContact = Identity & Contact;

// A contact can be either business or personal
type AllContacts = BusinessContact | PersonalContact;

Type Aliases

Type aliases create a new name for a type. They can be used to simplify complex type definitions:

// Simple type alias
type ID = string | number;

// Complex type alias
type UserData = {
    id: ID;
    name: string;
    preferences: {
        theme: "light" | "dark";
        notifications: boolean;
    };
    roles: string[];
};

// Usage
const user: UserData = {
    id: "user123",
    name: "Alice",
    preferences: {
        theme: "dark",
        notifications: true
    },
    roles: ["admin", "editor"]
};

Literal Types

Literal types allow you to specify exact values that a variable can have:

// String literal types
type Direction = "north" | "south" | "east" | "west";
let direction: Direction = "north";  // Valid
// direction = "northeast";  // Error: Type '"northeast"' is not assignable to type 'Direction'

// Numeric literal types
type DiceRoll = 1 | 2 | 3 | 4 | 5 | 6;
let roll: DiceRoll = 6;  // Valid
// roll = 7;  // Error: Type '7' is not assignable to type 'DiceRoll'

// Boolean literal types (less common but possible)
type True = true;
let t: True = true;  // Valid
// t = false;  // Error: Type 'false' is not assignable to type 'true'

Discriminated Unions

Discriminated unions (also called tagged unions) are a pattern where a common property is used to differentiate between union members:

// Shapes with a discriminant property 'kind'
interface Circle {
    kind: "circle";
    radius: number;
}

interface Square {
    kind: "square";
    sideLength: number;
}

interface Rectangle {
    kind: "rectangle";
    width: number;
    height: number;
}

// Union type
type Shape = Circle | Square | Rectangle;

// Function that uses the discriminant to handle each shape
function calculateArea(shape: Shape): number {
    switch (shape.kind) {
        case "circle":
            return Math.PI * shape.radius ** 2;
        case "square":
            return shape.sideLength ** 2;
        case "rectangle":
            return shape.width * shape.height;
    }
}

Exhaustiveness Checking

You can use TypeScript's never type to ensure all cases are handled:

function calculateArea(shape: Shape): number {
    switch (shape.kind) {
        case "circle":
            return Math.PI * shape.radius ** 2;
        case "square":
            return shape.sideLength ** 2;
        case "rectangle":
            return shape.width * shape.height;
        default:
            // This will cause a compile-time error if a new shape is added
            // and not handled above
            const exhaustiveCheck: never = shape;
            return exhaustiveCheck;
    }
}

Index Types

Index types allow you to create types based on the property names of another type:

interface Person {
    name: string;
    age: number;
    address: string;
    email: string;
}

// keyof operator - creates a union of property names
type PersonKeys = keyof Person;  // "name" | "age" | "address" | "email"

// Indexed access operator - gets the type of a property
type AgeType = Person["age"];  // number

// Generic function that safely accesses properties
function getProperty(obj: T, key: K): T[K] {
    return obj[key];
}

const person: Person = {
    name: "Alice",
    age: 30,
    address: "123 Main St",
    email: "alice@example.com"
};

// Type-safe property access
const name = getProperty(person, "name");  // string
const age = getProperty(person, "age");    // number
// const invalid = getProperty(person, "invalid");  // Error: Argument of type '"invalid"' is not assignable to parameter of type 'keyof Person'

Mapped Types

Mapped types allow you to create new types by transforming properties of existing types:

interface Person {
    name: string;
    age: number;
    email: string;
}

// Make all properties optional
type PartialPerson = {
    [K in keyof Person]?: Person[K];
};

// Make all properties readonly
type ReadonlyPerson = {
    readonly [K in keyof Person]: Person[K];
};

// Change property types
type PersonWithFlags = {
    [K in keyof Person]: Person[K] | boolean;
};

// Usage
const partialPerson: PartialPerson = { name: "Bob" };  // Valid, other properties are optional
const readonlyPerson: ReadonlyPerson = { name: "Charlie", age: 25, email: "charlie@example.com" };
// readonlyPerson.age = 26;  // Error: Cannot assign to 'age' because it is a read-only property

Conditional Types

Conditional types select a type based on a condition, similar to a ternary operator for types:

// Basic conditional type
type IsString = T extends string ? true : false;

type A = IsString;  // true
type B = IsString;  // false

// More complex example
type ArrayElementType = T extends (infer E)[] ? E : never;

type C = ArrayElementType;  // string
type D = ArrayElementType;  // number
type E = ArrayElementType;   // never (not an array)

// Distributive conditional types
type ToArray = T extends any ? T[] : never;

type F = ToArray;  // string[] | number[]

// Non-distributive version using square brackets
type ToArrayNonDist = [T] extends [any] ? T[] : never;

type G = ToArrayNonDist;  // (string | number)[]

Template Literal Types

Template literal types allow you to create new string types by concatenating other types:

// Basic template literal type
type Greeting = `Hello, ${string}!`;
let greeting: Greeting = "Hello, World!";  // Valid
let invalid: Greeting = "Hi there";  // Error: Type '"Hi there"' is not assignable to type '`Hello, ${string}!`'

// Combining with union types
type Direction = "top" | "right" | "bottom" | "left";
type Margin = `margin-${Direction}`;  // "margin-top" | "margin-right" | "margin-bottom" | "margin-left"

// Multiple substitutions
type Alignment = "start" | "end" | "center";
type JustifyContent = `justify-content-${Alignment}`;  // "justify-content-start" | "justify-content-end" | "justify-content-center"
type AlignItems = `align-items-${Alignment}`;  // "align-items-start" | "align-items-end" | "align-items-center"

type FlexAlignment = JustifyContent | AlignItems;  // Union of all possible alignments

// With numbers
type Padding = 0 | 1 | 2 | 3 | 4 | 5;
type PaddingClass = `p-${Padding}` | `pt-${Padding}` | `pb-${Padding}` | `pl-${Padding}` | `pr-${Padding}`;

// Usage
let paddingClass: PaddingClass = "p-3";  // Valid
// let invalidPadding: PaddingClass = "p-6";  // Error: Type '"p-6"' is not assignable to type 'PaddingClass'

The 'infer' Keyword

The infer keyword allows you to extract types from other types within conditional types:

// Extract the return type of a function
type ReturnType = T extends (...args: any[]) => infer R ? R : any;

function greet(name: string): string {
    return `Hello, ${name}!`;
}

type GreetReturn = ReturnType;  // string

// Extract parameter types
type FirstParameter = T extends (first: infer P, ...args: any[]) => any ? P : never;

function createUser(name: string, age: number, isAdmin: boolean) {
    return { name, age, isAdmin };
}

type FirstParam = FirstParameter;  // string

// Extract array element type
type ElementType = T extends (infer E)[] ? E : never;

type NumberArrayElement = ElementType;  // number
type StringArrayElement = ElementType;  // string

// Extract promise value type
type PromiseValue = T extends Promise ? V : never;

type ResolvedValue = PromiseValue>;  // string

Recursive Types

TypeScript supports recursive types, which are types that refer to themselves:

// Recursive type for JSON values
type JSONValue = 
    | string
    | number
    | boolean
    | null
    | JSONArray
    | JSONObject;

interface JSONArray extends Array {}

interface JSONObject {
    [key: string]: JSONValue;
}

// Usage
const data: JSONValue = {
    name: "Product",
    price: 29.99,
    inStock: true,
    tags: ["electronics", "gadgets"],
    dimensions: {
        width: 10,
        height: 5,
        unit: "cm"
    },
    related: null
};

// Recursive type for nested file system
type FileSystemItem = File | Directory;

interface File {
    type: "file";
    name: string;
    size: number;
}

interface Directory {
    type: "directory";
    name: string;
    children: FileSystemItem[];
}

// Usage
const fileSystem: FileSystemItem = {
    type: "directory",
    name: "root",
    children: [
        {
            type: "file",
            name: "readme.txt",
            size: 1024
        },
        {
            type: "directory",
            name: "src",
            children: [
                {
                    type: "file",
                    name: "index.ts",
                    size: 2048
                }
            ]
        }
    ]
};
